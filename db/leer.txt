--Tabla para los roles 
CREATE TABLE cargos (
  id INT AUTO_INCREMENT PRIMARY KEY,
  rol VARCHAR(50)
);

--Tabla para los usuarios
CREATE TABLE usuarios (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(100),
  usuario VARCHAR(50) UNIQUE,
  contrase単a VARCHAR(50),
  id_cargo INT,
  email VARCHAR(100),
  FOREIGN KEY (id_cargo) REFERENCES cargos (id)
);
